[ios] Fix IWYU issues with components/signin/public/base/consent_level.h

This prevents IWYU issues in an upcoming change that removes
consent_level.h imports from some headers. This change is correct
because:

files=$(git diff @{u} --name-only -- '*.h' '*.cc' '*.mm')
# It touches 125 C++/ObjC files.
echo "$files" | wc -l
# All of those contain "signin::ConsentLevel".
echo "$files" | xargs git grep -l signin::ConsentLevel | wc -l
# None of those contains "signin::ConsentLevel" in a comment.
echo "$files" | xargs git grep -El '//.*signin::ConsentLevel' | wc -l
